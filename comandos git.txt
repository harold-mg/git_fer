git --version (saber si tienes git instaldo)

git config --global user.name
git config --global user.email

---------SUBIR UN PROYECTO----------
git init
git add.
git status (ver los cambios)
git it remote add origin https://github.com...
git commit -m "comentario"
git push -u origin master

----------ACTUALIZAR UN PROYECTO--------
git status
git add .
git commit -m "comentario"
git push

---------CREAR BRANCH-------
git branch (ver la rama actual)
git branch nombre_rama     (crear rama)
git checkout nombre_rama   (moverse a la rama)
